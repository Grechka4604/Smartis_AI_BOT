Smartcallback генерирует определённые события, за которые может "зацепиться" клиент, чтобы выполнять необходимые действия с помощью кода


//При создании заявки создается событие SCBQueryCreated
document.addEventListener("SCBQueryCreated", function (event) {
//Your code here
});


//При открытии виджета создается событие SCBWidgetOpen
document.addEventListener("SCBWidgetOpen", function (event) {
//Your code here
});


//При заявке с scbForms создается событие SCBDirectCallFormSended
document.addEventListener("SCBDirectCallFormSended", function (event) {
//Your code here
});


//по закрытию создается событие SCBWidgetClose
document.addEventListener("SCBWidgetClose", function (event) {
//Your code here
});
